    function builtinRead(x) {
      if (Sk.builtinFiles === undefined || Sk.builtinFiles['files'][x] === undefined) {
        throw `File not found: '${x}'`;
      }
      return Sk.builtinFiles['files'][x];
    }

    async function executePython(code) {
      let output = '';
      Sk.pre = 'output';
      Sk.configure({
        output: function(text) {
          output += text;
        },
        read: builtinRead,
        __future__: Sk.python3
      });

      try {
        await Sk.misceval.asyncToPromise(function() {
          return Sk.importMainWithBody('<stdin>', false, code, true);
        });
        return { success: true, output: output.replace(/\n$/, '') };
      } catch (err) {
        return { success: false, output: String(err) };
      }
    }

    async function runCode() {
      const code = document.getElementById('code').value;
      setStatus('코드를 실행했습니다.', 'info');
      setOutput('실행 중...');
      const result = await executePython(code);
      if (result.success) {
        setOutput(result.output || '(출력 없음)');
      } else {
        setStatus('실행 중 오류가 발생했습니다.', 'fail');
        setOutput(result.output);
      }
    }

    async function checkAnswer() {
      const code = document.getElementById('code').value;
      setStatus('정답을 확인하는 중입니다...', 'info');
      setOutput('채점 중...');

      const result = await executePython(code);

      if (!result.success) {
        latestCheckResult = '오답';
        setStatus('오답입니다. 먼저 오류를 해결하세요.', 'fail');
        setOutput(`실행 오류\n\n${result.output}`);
        return;
      }

      if (result.output === expectedOutput) {
        latestCheckResult = '정답';
        setStatus('정답입니다! 출력이 예시와 완전히 같습니다.', 'pass');
        setOutput(`실행 결과\n\n${result.output}\n\n채점 결과\n정답`);
      } else {
        latestCheckResult = '오답';
        const expectedLines = expectedOutput.split('\n');
        const actualLines = result.output.split('\n');
        let feedback = '실행 결과\n\n' + (result.output || '(출력 없음)') + '\n\n채점 결과\n오답\n\n';
        feedback += '비교 안내\n';
        const maxLen = Math.max(expectedLines.length, actualLines.length);
        for (let i = 0; i < maxLen; i++) {
          const expected = expectedLines[i] ?? '(없음)';
          const actual = actualLines[i] ?? '(없음)';
          if (expected === actual) {
            feedback += `${i + 1}번째 줄: 일치\n`;
          } else {
            feedback += `${i + 1}번째 줄: 다름\n- 정답: ${expected}\n- 작성: ${actual}\n`;
          }
        }
        setStatus('오답입니다. 출력 내용을 다시 확인하세요.', 'fail');
        setOutput(feedback);
      }
    }

    function resetCode() {
